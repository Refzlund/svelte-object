	{#snippet item(prop)}
		<I.Object name={prop.index}>
			<Input name='name' min={2}>Pets name</Input>
			<Input name='age' type='number'>Pets age</Input>
		</I.Object>
	{/snippet}
	<button on:click={() => pets.push({})}> Add pet </button>
</I.Array>
